Basic Presentation Creation

from camel.toolkits import PPTXToolkit
import json

toolkit = PPTXToolkit(working_directory="./presentations")

# Define presentation content
content = [
    {
        "title": "CAMEL-AI Overview",
        "subtitle": "Multi-Agent Framework"
    },
    {
        "heading": "Key Features",
        "bullet_points": [
            "**Multi-agent** collaboration",
            "*Powerful* toolkits integration",
            "Open source and community-driven"
        ],
        "img_keywords": "artificial intelligence technology"
    }
]

# Create presentation
result = toolkit.create_presentation(
    content=json.dumps(content),
    filename="camel_presentation.pptx"
)

print(result)

Using with ChatAgent

from camel.agents import ChatAgent
from camel.models import ModelFactory
from camel.toolkits import PPTXToolkit
from camel.types import ModelPlatformType, ModelType

model = ModelFactory.create(
    model_platform=ModelPlatformType.OPENAI,
    model_type=ModelType.GPT_4O,
)

pptx_toolkit = PPTXToolkit(working_directory="./pptx_outputs")

agent = ChatAgent(
    system_message="You create professional PowerPoint presentations.",
    model=model,
    tools=pptx_toolkit.get_tools(),
)

task = """
Create a presentation about CAMEL-AI with:
1. Title slide
2. Introduction slide with key features
3. Use cases slide
4. Step-by-step workflow
5. Summary with a comparison table
"""

response = agent.step(task)
print(response.msgs[0].content)

import json
from camel.toolkits import PPTXToolkit

toolkit = PPTXToolkit(working_directory="./presentations")

presentation_content = [
    # Title slide
    {
        "title": "Modern Software Development",
        "subtitle": "Best Practices and Tools"
    },
    
    # Bullet points with image
    {
        "heading": "Key Development Principles",
        "bullet_points": [
            "**Agile** methodology for flexible development",
            "*Continuous Integration* and *Continuous Deployment*",
            "**Test-Driven Development** (TDD)",
            "Code review and pair programming"
        ],
        "img_keywords": "software development team"
    },
    
    # Step-by-step process
    {
        "heading": "Step-by-Step: Development Workflow",
        "bullet_points": [
            ">> **Step 1:** Plan and gather requirements",
            ">> **Step 2:** Design system architecture",
            ">> **Step 3:** Implement features",
            ">> **Step 4:** Write and run tests",
            ">> **Step 5:** Deploy and monitor"
        ],
        "img_keywords": "workflow diagram"
    },
    
    # Table
    {
        "heading": "Development Approaches Comparison",
        "table": {
            "headers": ["Approach", "Pros", "Cons"],
            "rows": [
                ["Waterfall", "Clear structure", "Less flexible"],
                ["Agile", "Flexible, iterative", "Requires discipline"],
                ["DevOps", "Fast delivery", "Complex setup"]
            ]
        }
    },
    
    # Summary slide
    {
        "heading": "Best Practices Summary",
        "bullet_points": [
            "Write clean, maintainable code",
            "Automate testing and deployment",
            "Monitor and optimize performance",
            "Document everything"
        ],
        "img_keywords": "https://example.com/success.jpg"
    }
]

# Create the presentation
result = toolkit.create_presentation(
    content=json.dumps(presentation_content),
    filename="modern_development.pptx",
    template="./templates/professional.pptx"
)

print(result)

ğŸ“ Best Practices

1. Use Templates for Branding

Create custom templates with your brand colors and fonts

2. Limit Bullet Points

Keep 3-5 bullet points per slide for better readability

3. Use Step-by-Step for Processes

Use >> prefix for process flows (3-6 steps work best)

4. Optimize Images

Use relevant keywords for automatic image selection

5. Validate JSON Structure

Always validate JSON before passing to create_presentation
