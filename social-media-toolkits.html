Quick Start

from camel.toolkits import RedditToolkit

toolkit = RedditToolkit(
    retries=3,
    delay=0.5
)

# Required environment variables:
# REDDIT_CLIENT_ID
# REDDIT_CLIENT_SECRET
# REDDIT_USER_AGENT

# Collect top posts
posts = toolkit.collect_top_posts(
    subreddit_name="artificialintelligence",
    post_limit=5,
    comment_limit=5
)

for post in posts:
    print(f"Post: {post['Post Title']}")
    for comment in post['Comments']:
        print(f"  - {comment['Comment Body']}")

Track Keywords

# Track specific keywords across subreddits
results = toolkit.track_keyword_discussions(
    subreddits=["MachineLearning", "artificial", "deeplearning"],
    keywords=["GPT", "transformer", "attention"],
    post_limit=10,
    comment_limit=10,
    sentiment_analysis=True
)

for item in results:
    print(f"Subreddit: {item['Subreddit']}")
    print(f"Comment: {item['Comment Body']}")
    print(f"Sentiment: {item['Sentiment Score']}")

Sentiment Analysis

# Perform sentiment analysis on collected data
data = toolkit.collect_top_posts("technology", 10, 10)
analyzed = toolkit.perform_sentiment_analysis(data)

# Sentiment scores: -1 (negative) to 1 (positive)

ğŸ“ Complete Example: Social Media Bot

from camel.agents import ChatAgent
from camel.toolkits import (
    TwitterToolkit,
    RedditToolkit,
    LinkedInToolkit
)

# Initialize toolkits
twitter = TwitterToolkit()
reddit = RedditToolkit()
linkedin = LinkedInToolkit()

# Create agent with all toolkits
agent = ChatAgent(
    system_message="""You are a social media manager bot.
    Monitor trends, create engaging content, and respond to mentions.""",
    tools=[
        *twitter.get_tools(),
        *reddit.get_tools(),
        *linkedin.get_tools(),
    ]
)

# Task: Monitor and post
task = """
1. Check trending topics on Reddit about AI
2. Create a summary tweet about the most interesting discussion
3. Share a professional post on LinkedIn about AI trends
"""

response = agent.step(task)
print(response.msgs[0].content)
